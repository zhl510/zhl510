  <script type="module">
    import * as THREE from 'three';
    import { OrbitControls } from 'three/addons/controls/OrbitControls.js';
    import { CSS2DRenderer, CSS2DObject } from 'three/addons/renderers/CSS2DRenderer.js';

    // --- 初始化场景、相机、渲染器 ---
    const container = document.getElementById('canvas-container');
    const scene = new THREE.Scene();
    scene.background = new THREE.Color(0x050e1a);
    scene.fog = new THREE.FogExp2(0x050e1a, 0.007);

    const camera = new THREE.PerspectiveCamera(42, container.clientWidth / container.clientHeight, 0.1, 1000);
    camera.position.set(5.8, 4.5, 6.2);
    camera.lookAt(0, 0, 0);

    const renderer = new THREE.WebGLRenderer({ antialias: true });
    renderer.setSize(container.clientWidth, container.clientHeight);
    renderer.shadowMap.enabled = true;
    renderer.setPixelRatio(window.devicePixelRatio);
    container.appendChild(renderer.domElement);
    
    const labelRenderer = new CSS2DRenderer();
    labelRenderer.setSize(container.clientWidth, container.clientHeight);
    labelRenderer.domElement.style.position = 'absolute';
    labelRenderer.domElement.style.top = '0px';
    labelRenderer.domElement.style.left = '0px';
    labelRenderer.domElement.style.pointerEvents = 'none';
    container.appendChild(labelRenderer.domElement);
    
    // 轨道控制
    const controls = new OrbitControls(camera, renderer.domElement);
    controls.enableDamping = true;
    controls.dampingFactor = 0.06;
    controls.autoRotate = false;
    controls.enableZoom = true;
    controls.enablePan = true;
    controls.zoomSpeed = 1.2;
    controls.rotateSpeed = 1.0;
    controls.target.set(0, 0, 0);
    
    // 辅助视觉: 极简网格
    const gridHelper = new THREE.GridHelper(12, 24, 0x88aaff, 0x446688);
    gridHelper.position.y = -0.9;
    gridHelper.material.transparent = true;
    gridHelper.material.opacity = 0.2;
    scene.add(gridHelper);
    
    const axesHelper = new THREE.AxesHelper(5);
    axesHelper.material.transparent = true;
    axesHelper.material.opacity = 0.06;
    scene.add(axesHelper);
    
    // 灯光系统
    const ambientLight = new THREE.AmbientLight(0x404064);
    scene.add(ambientLight);
    const mainLight = new THREE.DirectionalLight(0xffffff, 1.2);
    mainLight.position.set(2, 5, 3);
    mainLight.castShadow = true;
    scene.add(mainLight);
    const fillLight = new THREE.PointLight(0x4488cc, 0.5);
    fillLight.position.set(-2.2, 1.8, 3.2);
    scene.add(fillLight);
    const backLight = new THREE.PointLight(0xffaa77, 0.45);
    backLight.position.set(0, 1.5, -4.8);
    scene.add(backLight);
    const rimLight = new THREE.PointLight(0xff9966, 0.4);
    rimLight.position.set(1.2, 1.2, -3.5);
    scene.add(rimLight);
    const hemiLight = new THREE.HemisphereLight(0x668899, 0x332c2a, 0.55);
    scene.add(hemiLight);
    
    // --- 杆件模型组 ---
    const rodGroup = new THREE.Group();
    scene.add(rodGroup);
    let currentCylinder = null;
    
    function getRadiusFromArea(area) { return Math.sqrt(area / Math.PI); }
    
    function updateRodVisual(visualLengthM, areaM2, intensityFactor = 0) {
      while(rodGroup.children.length > 0) {
        rodGroup.remove(rodGroup.children[0]);
      }
      
      const radiusVal = getRadiusFromArea(areaM2);
      const finalRadius = Math.max(radiusVal, 0.085);
      // 形变强度决定颜色：从冷色(0.55)到暖红(0.2)
      const hue = 0.55 - intensityFactor * 0.38;
      const sat = 0.7 + intensityFactor * 0.25;
      const light = 0.55 + intensityFactor * 0.12;
      const materialColor = new THREE.Color().setHSL(hue, sat, light);
      
      const material = new THREE.MeshStandardMaterial({ 
        color: materialColor, 
        roughness: 0.28, 
        metalness: 0.82,
        emissive: new THREE.Color().setHSL(hue, 0.9, 0.1),
        emissiveIntensity: 0.1 + intensityFactor * 0.5,
        flatShading: false
      });
      const geometry = new THREE.CylinderGeometry(finalRadius, finalRadius, visualLengthM, 96, 64);
      const cylinder = new THREE.Mesh(geometry, material);
      cylinder.castShadow = true;
      cylinder.rotation.z = Math.PI / 2;
      cylinder.position.set(0, 0, 0);
      rodGroup.add(cylinder);
      currentCylinder = cylinder;
      
      // 端帽金属质感
      const capMaterial = new THREE.MeshStandardMaterial({ color: 0xdd9955, metalness: 0.94, roughness: 0.18 });
      const capGeo = new THREE.SphereGeometry(finalRadius * 0.94, 48, 36);
      const leftCap = new THREE.Mesh(capGeo, capMaterial);
      const rightCap = new THREE.Mesh(capGeo, capMaterial);
      leftCap.position.set(-visualLengthM / 2, 0, 0);
      rightCap.position.set(visualLengthM / 2, 0, 0);
      leftCap.castShadow = true;
      rightCap.castShadow = true;
      rodGroup.add(leftCap);
      rodGroup.add(rightCap);
    }
    
    // 拉力箭头与标签
    let arrowsAndLabels = null;
    function initArrows() {
      const arrowColor = 0xff7777;
      const arrowRight = new THREE.ArrowHelper(new THREE.Vector3(1,0,0), new THREE.Vector3(1.5,0,0), 0.82, arrowColor, 0.45, 0.32);
      const arrowLeft = new THREE.ArrowHelper(new THREE.Vector3(-1,0,0), new THREE.Vector3(-1.5,0,0), 0.82, arrowColor, 0.45, 0.32);
      rodGroup.add(arrowRight);
      rodGroup.add(arrowLeft);
      
      const leftDiv = document.createElement('div');
      leftDiv.textContent = '← 拉力 F';
      leftDiv.style.cssText = 'color:#ffbb99;font-size:14px;font-weight:bold;background:rgba(0,0,0,0.6);padding:2px 12px;border-radius:28px;border-left:3px solid #ff8866;backdrop-filter:blur(4px);';
      const rightDiv = document.createElement('div');
      rightDiv.textContent = '拉力 F →';
      rightDiv.style.cssText = 'color:#ffbb99;font-size:14px;font-weight:bold;background:rgba(0,0,0,0.6);padding:2px 12px;border-radius:28px;border-right:3px solid #ff8866;backdrop-filter:blur(4px);';
      const leftLabel = new CSS2DObject(leftDiv);
      const rightLabel = new CSS2DObject(rightDiv);
      rodGroup.add(leftLabel);
      rodGroup.add(rightLabel);
      return { arrowRight, arrowLeft, leftLabel, rightLabel };
    }
    
    arrowsAndLabels = initArrows();
    function updateArrowsPosition(len) {
      const half = len / 2;
      arrowsAndLabels.arrowRight.position.set(half + 0.28, 0, 0);
      arrowsAndLabels.arrowLeft.position.set(-half - 0.28, 0, 0);
      arrowsAndLabels.leftLabel.position.set(-half - 0.78, 0.45, 0);
      arrowsAndLabels.rightLabel.position.set(half + 0.78, 0.45, 0);
    }
    
    // 星空背景点缀
    const starFieldGeo = new THREE.BufferGeometry();
    const starCount = 1300;
    const starPos = new Float32Array(starCount * 3);
    for (let i=0; i<starCount; i++) {
      starPos[i*3] = (Math.random() - 0.5) * 480;
      starPos[i*3+1] = (Math.random() - 0.5) * 260;
      starPos[i*3+2] = (Math.random() - 0.5) * 200 - 80;
    }
    starFieldGeo.setAttribute('position', new THREE.BufferAttribute(starPos, 3));
    const starMat = new THREE.PointsMaterial({ color: 0xffffff, size: 0.07, transparent: true, opacity: 0.45 });
    const stars = new THREE.Points(starFieldGeo, starMat);
    scene.add(stars);
    
    // --- 物理计算与UI绑定 ---
    const forceSlider = document.getElementById('force');
    const lengthSlider = document.getElementById('length');
    const areaSlider = document.getElementById('area');
    const modulusSlider = document.getElementById('modulus');
    const forceValSpan = document.getElementById('forceVal');
    const lengthValSpan = document.getElementById('lengthVal');
    const areaValSpan = document.getElementById('areaVal');
    const modulusValSpan = document.getElementById('modulusVal');
    const deformationSpan = document.getElementById('deformation');
    const stressSpan = document.getElementById('stress');
    const strainSpan = document.getElementById('strain');
    const hookeCheckSpan = document.getElementById('hookeCheck');
    
    let F = 500, L = 2.0, A = 0.01, E_GPa = 200, E = 200e9;
    function computeDeltaL(force, length, area, Epa) { return (force * length) / (Epa * area); }
    
    function updateAll() {
      E = E_GPa * 1e9;
      const deltaL = computeDeltaL(F, L, A, E);
      const strainVal = deltaL / L;
      const stressPa = F / A;
      const stressMPa = stressPa / 1e6;
      // 视觉强度因子: 基于相对伸长映射，限制最大0.85
      let intensity = Math.min(0.85, Math.abs(deltaL) / (L * 0.18));
      intensity = Math.max(0.06, Math.pow(intensity, 0.9));
      
      deformationSpan.textContent = deltaL.toFixed(8);
      stressSpan.textContent = stressMPa.toFixed(3);
      strainSpan.textContent = strainVal.toExponential(5);
      hookeCheckSpan.innerHTML = `${strainVal.toExponential(5)}  ≈  ${(stressPa/E).toExponential(5)}`;
      
      let visualLength = L + deltaL;
      visualLength = Math.max(visualLength, 0.48);
      updateRodVisual(visualLength, A, intensity);
      updateArrowsPosition(visualLength);
    }
    
    function syncFromUI() {
      F = parseFloat(forceSlider.value);
      L = parseFloat(lengthSlider.value);
      A = parseFloat(areaSlider.value);
      E_GPa = parseFloat(modulusSlider.value);
      forceValSpan.textContent = F.toFixed(1);
      lengthValSpan.textContent = L.toFixed(3);
      areaValSpan.textContent = A.toFixed(4);
      modulusValSpan.textContent = E_GPa.toFixed(1);
      updateAll();
    }
    
    forceSlider.addEventListener('input', syncFromUI);
    lengthSlider.addEventListener('input', syncFromUI);
    areaSlider.addEventListener('input', syncFromUI);
    modulusSlider.addEventListener('input', syncFromUI);
    syncFromUI();
    
    // 操控台的收起/展开逻辑
    const dashboard = document.getElementById('dashboard');
    const handle = document.getElementById('dashboardHandle');
    handle.addEventListener('click', (e) => {
      e.stopPropagation();
      dashboard.classList.toggle('collapsed');
    });
    
    // 简单动画: 星星缓慢旋转
    let timeAnim = 0;
    function animate() {
      requestAnimationFrame(animate);
      timeAnim += 0.016;
      stars.rotation.y = timeAnim * 0.0007;
      stars.rotation.x = Math.sin(timeAnim * 0.0004) * 0.04;
      controls.update();
      renderer.render(scene, camera);
      labelRenderer.render(scene, camera);
    }
    animate();
    
    window.addEventListener('resize', () => {
      const w = container.clientWidth, h = container.clientHeight;
      camera.aspect = w / h;
      camera.updateProjectionMatrix();
      renderer.setSize(w, h);
      labelRenderer.setSize(w, h);
    });
    
    console.log('✅ 极简形变可视化 | 可收起操控台 | 杆体颜色/长度随胡克定律实时变化');
  </script>
